function setupInitialSheets() {
  var doc = SpreadsheetApp.getActiveSpreadsheet();
  var sheetsToCreate = [
    { name: 'Items', headers: ['id', 'name', 'sku', 'description', 'category', 'type', 'uom', 'reorderLevel', 'unitPrice'] },
    { name: 'Departments', headers: ['id', 'name', 'head', 'location'] },
    { name: 'Warehouses', headers: ['id', 'name', 'location', 'type'] },
    { name: 'Suppliers', headers: ['id', 'name', 'contactPerson', 'email', 'phone', 'address', 'gstin'] },
    { name: 'GateEntries', headers: ['id', 'entryNo', 'date', 'type', 'partyName', 'vehicleNo', 'driverName', 'remarks', 'status'] },
    { name: 'MaterialIssues', headers: ['id', 'issueNo', 'date', 'departmentId', 'requestedBy', 'issuedBy', 'status', 'remarks'] },
    { name: 'PRs', headers: ['id', 'prNo', 'date', 'departmentId', 'requestedBy', 'expectedDate', 'status', 'remarks'] },
    { name: 'POs', headers: ['id', 'poNo', 'date', 'supplierId', 'expectedDelivery', 'status', 'totalAmount', 'remarks'] },
    { name: 'GRNs', headers: ['id', 'grnNo', 'poId', 'gateEntryId', 'date', 'receivedBy', 'inspectedBy', 'status', 'remarks'] },
    { name: 'StockTransfers', headers: ['id', 'transferNo', 'date', 'fromWarehouseId', 'toWarehouseId', 'requestedBy', 'status', 'remarks'] },
    { name: 'StockAdjustments', headers: ['id', 'adjustmentNo', 'date', 'warehouseId', 'adjustedBy', 'reason', 'status', 'remarks'] }
  ];

  sheetsToCreate.forEach(function(s) {
    var sheet = doc.getSheetByName(s.name);
    if (!sheet) {
      sheet = doc.insertSheet(s.name);
    }
    // Check if headers exist
    var lastRow = sheet.getLastRow();
    if (lastRow === 0) {
      sheet.appendRow(s.headers);
      sheet.getRange(1, 1, 1, s.headers.length).setFontWeight("bold");
    }
  });
}

function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      throw new Error("No data received");
    }

    // Parse the payload correctly handling CORS text/plain fallback
    var payload = JSON.parse(e.postData.contents);
    var action = payload.action;
    var sheetName = payload.sheetName;
    var data = payload.data;
    var spreadsheetId = payload.spreadsheetId;
    
    var doc;
    if (spreadsheetId) {
      doc = SpreadsheetApp.openById(spreadsheetId);
    } else {
      doc = SpreadsheetApp.getActiveSpreadsheet();
    }
    
    if (!doc) {
      throw new Error("Could not find active spreadsheet. Please provide a Spreadsheet ID or link the script to a Google Sheet.");
    }
    
    // Map sheet names
    var sheetNameMapping = {
      'items': 'Items',
      'departments': 'Departments',
      'warehouses': 'Warehouses',
      'suppliers': 'Suppliers',
      'gate_entries': 'GateEntries',
      'material_issues': 'MaterialIssues',
      'material_issue_items': 'MaterialIssues',
      'prs': 'PRs',
      'pos': 'POs',
      'grns': 'GRNs',
      'stock_transfers': 'StockTransfers',
      'stock_adjustments': 'StockAdjustments'
    };
    
    var targetSheetName = sheetNameMapping[sheetName] || sheetName;
    var sheet = doc.getSheetByName(targetSheetName);
    
    if (!sheet) {
      sheet = doc.insertSheet(targetSheetName);
    }
    
    if (action === 'append') {
      var inputKeys = Object.keys(data);
      var rowData = [];
      var headers = [];
      
      var lastRow = sheet.getLastRow();
      var lastCol = sheet.getLastColumn();
      
      // Auto-generate or update headers
      if (lastRow === 0 || lastCol === 0) {
        // Sheet is completely empty, write headers to first row
        headers = inputKeys;
        sheet.appendRow(headers);
        sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");
      } else {
        // Read existing headers
        headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
        
        var newHeadersAdded = false;
        for (var i = 0; i < inputKeys.length; i++) {
          if (headers.indexOf(inputKeys[i]) === -1) {
            headers.push(inputKeys[i]);
            newHeadersAdded = true;
          }
        }
        
        // Update header row if there are new headers
        if (newHeadersAdded) {
          sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
          sheet.getRange(1, 1, 1, headers.length).setFontWeight("bold");
        }
      }
      
      // Map data to the correct column order based on headers
      for (var j = 0; j < headers.length; j++) {
        var key = headers[j];
        var val = data[key];
        
        if (val === null || val === undefined) {
          rowData.push("");
        } else if (typeof val === 'object') {
          rowData.push(JSON.stringify(val));
        } else {
          rowData.push(val);
        }
      }
      
      sheet.appendRow(rowData);
    }
    
    return ContentService.createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString(), trace: error.stack }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

}
